      city: "Manchester",
      postcode: "M1 7AJ",
      tenant: "John Cena",

            <th className="py-2 px-4">Status</th>
            <th className="py-2 px-4">Actions</th> {/* Actions column */}
          </tr>
        </thead>
        <tbody>
          {properties.map((property) => (
